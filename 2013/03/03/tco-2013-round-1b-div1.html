250pt EllysPairs

排序一下求最大最小即可。

 1 #include <cstdio>
 2 #include <iostream>
 3 #include <cstring>
 4 #include <algorithm>
 5 #include <cmath>
 6 #include <vector>
 7 using namespace std ;
 8 #define For(i , n) for(int i = 0 ; i < (n) ; ++i)
 9 #define SZ(x)  (int)((x).size())
10 typedef long long lint ;
11 const int maxint = -1u>>2 ;
12 const double eps = 1e-6 ; 
13 class EllysPairs
14 {
15 public:
16 int getDifference(vector <int> knowledge)
17 {
18     int tmp;
19     vector <int> g;
20     g.clear();
21     sort(knowledge.begin(), knowledge.end());
22     for(int i = 0; i < SZ(knowledge) / 2; i ++) {
23         tmp = knowledge[i] + knowledge[SZ(knowledge) - i - 1];
24         g.push_back(tmp);
25     }
26     sort(g.begin(), g.end());
27     return int(g[SZ(g) - 1] - g[0]);
28 }
29 
30 };

500pt EllysFigurines

枚举行或者列的所有操作，然后再去判断列或者行的操作。

17 public:

20 
21 void init(vector <string> board) {
22     for(int i = 0; i < SZ(board); i ++) {
23         for(int j = 0; j < SZ(board[0]); j ++) {

29 int getMoves(vector <string> board, int R, int C) {
30     bool flag;
31     int m[40]; 
32     int ans = 40000, tans, t;
33     for(int i = 0; i < 1 << (SZ(board) + 1); i ++) {
34         tans = 0;
35         init(board);
36         t = 0;
37         while(t < SZ(board)) {
38             int tmp = 1 << t;
39             if((i & tmp) > 0) {
40                 for(int j = t; j < t + R; j ++) {
41                     for(int k = 0; k < SZ(board[0]); k ++) {
42                         maze[j][k] = '.';
43                     }
44                 }
45                 t += R;
46                 tans ++;
47             } 
48             else {
49                 t ++;
50             }
51         }
52         memset(m, 0, sizeof(m));
53         for(int j = 0; j < SZ(board[0]); j ++) {
54             flag = true;
55             for(int k = 0; k < SZ(board); k ++) {
56                 if(maze[k][j] == 'X') {
57                     flag = false;
58                     break;
59                 }
60             }
61             if(!flag) {
62                 m[j] = 1;
63             }
64         }
65         t = 0;

67             if(m[t]) {
68                 t += C;
69                 tans ++;
70             }
71             else {
72                 t ++;
73             }
74         }

1000pt EllysReversals

每次只能逆转前偶数个元素，就说明每两个字母可以列一组，组内可以随意交换，组间可以随意交换，将每个字符串用字典序排序即可。

注意如果是字母总和如果是奇数，最后一个字母不能改变位置。

 1 #include <cstdio>
 2 #include <iostream>
 3 #include <cstring>
 4 #include <algorithm>
 5 #include <cmath>
 6 #include <vector>
 7 using namespace std ;
 8 #define For(i , n) for(int i = 0 ; i < (n) ; ++i)
 9 #define SZ(x)  (int)((x).size())
10 typedef long long lint ;
11 const int maxint = -1u>>2 ;
12 const double eps = 1e-6 ; 
13 
14 class EllysReversals
15 {
16 public:
17 int getMin(vector <string> words)
18 {
19     int ans = 0;
20     bool flag[60];
21     string tmp;
22     vector <string> rec[60];
23     memset(flag, true, sizeof(flag));
24     for(int i = 0; i < SZ(words); i ++) {
25         rec[i].clear();
26         for(int j = 0; j < SZ(words[i]); j += 2) { 
27             tmp = "";
28             tmp += words[i][j];
29             if((j + 1) < SZ(words[i]))
30                 tmp += words[i][j + 1];
31             sort(tmp.begin(), tmp.end());
32             if(SZ(tmp) == 2)
33                 rec[i].push_back(tmp);
34         }
35         sort(rec[i].begin(), rec[i].end());
36         if(SZ(words[i]) % 2 != 0) {
37             rec[i].push_back(tmp);
38         }
39         words[i] = "";
40         for(int k = 0; k < SZ(rec[i]); k ++) {
41             words[i] += rec[i][k];
42         }
43     }
44     for(int i = 0; i < SZ(words); i ++) {
45         for(int j = i + 1; j < SZ(words); j ++) {
46             if(words[i] == words[j] && flag[i] && flag[j]) {
47                 flag[i] = flag[j] = false;
48                 ans ++;
49             }
50         }
51     }
52     ans = SZ(words) - ans * 2;
53     return int(ans);
54 }
55 
56 };
