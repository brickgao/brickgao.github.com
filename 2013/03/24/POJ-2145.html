置换群，burnside引理的题目。

这道题代表了一类题目的优化

裸的算法是 ∑ngcd(n,i) 1<i<=n

复杂度过高，进行优化。

置换群种循环的个数L = n / gcd(n, i)

因为如果L | n, 则有n / L | n

则环的长度L的范围是1 ~ sqrt(L)

令a = gcd(n, i), 设i = at

则只有i与t互质的时候，gcd(i, t) = a

则最后可以优化为 ∑(Φ(i) * n ^ (i)) % p

22 void getprime() {
23     num = 0;
24     for(int i = 2; i <= 50000; i ++) {
25         if(!isprime[i]) {
26             prime[num ++] = i;
27             for(int j = 1; j * i <= 50000; j ++) {
28                 isprime[i * j] = 1;
29             }
30         }
31     }
32 }

34 int euler(int x) {
35     int res = x;
36     for(int i = 0; i < num && prime[i] * prime[i] <= x; i++) {
37         if(x % prime[i] == 0) {
38             res = res / prime[i] * (prime[i] - 1);
39             while(x % prime[i] == 0) {
40                 x /= prime[i];
41             }
42         }
43     }
44     if(x > 1) res = res / x * (x - 1);
45     return res;
46 }

48 LL quickpow(LL m , LL n , LL k) {
49     LL tmp = 1; 
50     m %= k;
51     while(n) { 
52         if(n & 1)
53             tmp = (tmp * m) % k; 
54         m = (m * m) % k;
55         n >>= 1;
56     } 
57     return tmp;
58 }

60 int main() {
61     getprime();
62     scanf("%d", &t);
63     while(t --) {
64         ans = 0;
65         scanf("%d%d", &n, &p);
66         for(int i = 1; i < sqrt(n); i ++) {
67             if(n % i == 0) {
68                 ans = (ans + euler(i) % p * quickpow(n, n / i - 1, p) + euler(n / i) % p * quickpow(n, i - 1, p)) % p;
69                 //cout << quickpow(n,n-1,p) << " " << quickpow(n, n / i - 1, p) << " " << euler(i) << " " << euler(n / i) << endl;
70             }
71         }

73             ans = (ans + quickpow(n, sqrt(n) - 1, p) * (euler(sqrt(n)) % p)) % p;
