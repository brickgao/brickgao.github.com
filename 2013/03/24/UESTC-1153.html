UESTC-1153 The Queen’s New Necklaces

2013-03-24 #Pólya定理 #置换群 #大数 #Burnside引理 #UVA

颜色限制的Burnside引理应用。

以为要用大数所以用java写了。

Code

 1 import java.io.*;
 2 import java.util.*;
 3 import java.math.*;
 4 import java.text.*;
 5 
 6 public class Main {
 7 
 8     static int gcd(int a, int b) {
 9         if(b == 0)  return a;
10         return gcd(b, a % b);
11     }
12 
13     public static void main(String[] args) throws Exception{
14         BigInteger f[] = new BigInteger[1005], ans;
15         int a[] = new int[110], b[] = new int[110], total;
16         Scanner cin = new Scanner(System.in);
17         int t, n;
18         f[0] = BigInteger.ONE;
19         for(int i = 1; i <= 1000; i++)             
20           f[i] = f[i - 1].multiply(BigInteger.valueOf(i));         
21           t = cin.nextInt();         
22           while(t -- > 0) {
23             ans = BigInteger.valueOf(0);
24             total = 0;
25             n = cin.nextInt();
26             for(int i = 0; i < n; i ++) {
27                 a[i] = cin.nextInt();
28                 total += a[i];
29             }
30             for(int i = 1; i <= total; i ++) {
31                 int flag = 1;
32                 int k = total / gcd(i, total);
33                 int nn = 0;
34                 for(int j = 0; j < n; j ++) {
35                     b[j] = a[j];
36                 }
37                 for(int j = 0; j < n; j ++) {
38                     if(b[j] % k == 0) {
39                         b[j] /= k;
40                         nn += b[j];
41                     }
42                     else {
43                         flag = 0;
44                         break;
45                     }
46                 }
47                 BigInteger sum = BigInteger.ONE;
48                 if(flag == 1) {
49                     //System.out.println(i);
50                     for(int j = 0; j < n; j ++) {
51                         sum = sum.multiply(f[nn]);
52                         sum = sum.divide(f[b[j]]);
53                         sum = sum.divide(f[nn - b[j]]);
54                         //System.out.println(sum);
55                         nn -= b[j];
56                     }
57                     ans = ans.add(sum);
58                 }
59             }
60             ans = ans.divide(BigInteger.valueOf(total));
61             System.out.println(ans);
62         }
63     }
64 }
