C. Rectangle Puzzle

找出一个分界点角度，即两个矩形不重叠，但对角线重叠，分两种情况算。

13 const double eps = 1e-8;
14 template <class T> bool get_max(T& a, const T &b) {return b > a? a = b, 1: 0;}

17 long long w, h, a, tmp1;
18 double x, ans, tmp, angel;
19 
20 int main() {
21     cin >> w >> h >> a;
22     if(h > w) {
23         tmp1 = h;
24         h = w;
25         w = tmp1;
26     }
27     if(a > 90) a = 180 - a;
28     if(a == 0) {
29         ans = (double)w * (double)h;
30         printf("%.9lf\n", ans);
31         return 0;
32     }
33     angel = (double)a / 180.0 * acos(-1);
34     x = (w * w - h * h) / (2 * w);
35     tmp = acos(x / (w - x));
36     if((double)angel > tmp - eps) {
37         ans = (double)h * (double)h / sin((double)angel);
38     }
39     else {
40         double t1 = (w * (1 + cos(angel)) - h * sin(angel)) / ((1 + cos(angel)) * (1 + cos(angel)) - sin(angel) * sin(angel));
41         double t2 = (w - (1 + cos(angel)) * t1) / sin(angel);
42         ans = (double)w * (double)h - (t1 * t1 + t2 * t2) * sin(angel) * cos(angel);
43     }
44     printf("%.9lf\n", ans);
45     return 0;
46 }

D. Maximum Xor Secondary

用的RMQ加二分做的，取每个点向左向右最近的一个比他大的值，做亦或运算...据说单调队列几行就能搞定。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 #define out(v) cerr << #v << ": " << (v) << endl
11 #define SZ(v) ((int)(v).size())
12 #define maxn 100001
13 const int maxint = -1u>>1;
14 template <class T> bool get_max(T& a, const T &b) {return b > a? a = b, 1: 0;}
15 template <class T> bool get_min(T& a, const T &b) {return b < a? a = b, 1: 0;}
16 
17 int n;
18 long long ans = - 1;
19 long long a[maxn];
20 long long dpmax[maxn][40];
21 
22 void Make_Big_RMQ(int n) {
23     for(int i = 1; i <= n; i ++)
24         dpmax[i][0] = a[i];
25     for(int j = 1; j <= log((double)n) / log(2.0); j ++) {
26         for(int i = 1; i + (1 << j) - 1 <= n; i ++) {
27             dpmax[i][j] = max(dpmax[i][j - 1], dpmax[i+(1<<(j-1))][j-1]);
28         }
29     }
30 }
31 
32 long long get_max_rmq(int a, int b) {
33     int k = (int)(log((double)(b - a + 1)) / log(2.0));
34     return max(dpmax[a][k],dpmax[b-(1<<k)+1][k]);
35 }
36 
37 int find1(int l, int r, int key) {
38     while(l != r && l != r - 1) {
39         if(get_max_rmq(key, (l + r) / 2) > a[key]) {
40             r = (l + r) / 2;
41         }
42         else {
43             l = (l + r) / 2;
44         }
45     }
46     if(l == r)  return l;
47     if(l == r - 1) {
48         if(get_max_rmq(key, l) > a[key])
49             return l;
50         if(get_max_rmq(key, r) > a[key])
51             return r;
52     }
53     return -1;
54 }
55 
56 int find2(int l, int r, int key) {
57     while(l != r && l != r - 1) {
58         if(get_max_rmq((l + r) / 2, key) > a[key]) {
59             l = (l + r) / 2;
60         }
61         else {
62             r = (l + r) / 2;
63         }
64     }
65     if(l == r)  return l;
66     if(l == r - 1) {
67         if(get_max_rmq(r, key) > a[key])
68             return r;
69         if(get_max_rmq(l, key) > a[key])
70             return l;
71     }
72     return -1;
73 }
74 
75 int main() {
76     cin >> n;
77     for(int i = 1; i <= n; i ++) {
78         cin >> a[i];
79     }
80     Make_Big_RMQ(n);
81     for(int i = 1; i <= n; i ++) {
82         if(get_max_rmq(i, n) > a[i]) {
83             int tmp1 = find1(i, n, i);
84             ans = max(ans, a[i] ^ a[tmp1]);
85         }
86         if(get_max_rmq(1, i) > a[i]) {
87             int tmp2 = find2(1, i, i);
88             ans = max(ans, a[i] ^ a[tmp2]);
89         }
90     }
91     cout << ans << endl;
92     return 0;
93 }

E. Game on Tree

根据期望的可加性，求出每个点被删除的期望相加，每个点的被删除概率是1/层数，乘以1就是其期望。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 #define out(v) cerr << #v << ": " << (v) << endl
11 #define SZ(v) ((int)(v).size())
12 const int maxint = -1u>>1;
13 template <class T> bool get_max(T& a, const T &b) {return b > a? a = b, 1: 0;}
14 template <class T> bool get_min(T& a, const T &b) {return b < a? a = b, 1: 0;}
15 
16 int n;
17 double ans;
18 vector <int> p[100010];
19 
20 void dfs(int u, int fa, int l) {
21     int v;
22     ans += 1.0 / (double)l;
23     for(int i = 0; i < SZ(p[u]); i ++) {
24         v = p[u][i];
25         if(v != fa) {
26             dfs(v, u, l + 1);
27         }
28     }
29 }
30 
31 int main() {
32     int u, v;
33     ans = 0.0;
34     cin >> n;
35     for(int i = 0; i <= n; i ++) {
36         p[i].clear();
37     }
38     for(int i = 0; i < n - 1; i ++) {
39         cin >> u >> v;
40         p[u].push_back(v);
41         p[v].push_back(u);
42     }
43     dfs(1, -1, 1);
44     printf("%.8lf\n", ans);
45     return 0;
46 }
