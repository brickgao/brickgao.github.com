最小矩阵乘积计算次数。

16 int n, p[110], s[110][110];
17 
18 int dp(int l, int r) {
19     if(s[l][r] > 0)
20         return s[l][r];
21     for(int i = l + 1; i < r; i ++) {
22         if(s[l][r] != 0)
23             s[l][r] = min(dp(l, i) + dp(i, r) + p[l] * p[i] * p[r], s[l][r]);
24         else {
25             s[l][r] = dp(l, i) + dp(i, r) + p[l] * p[i] * p[r];
26         }
27     }
28     return s[l][r];
29 }
30 
31 int main() {
32     while(scanf("%d", &n) != EOF) {
33         memset(s, 0, sizeof(s));
34         for(int i = 0; i < n; i ++) {
35             scanf("%d", &p[i]);
36         }
37         dp(0, n - 1);
38         printf("%d\n", s[0][n - 1]);
39     }
40     return 0;
