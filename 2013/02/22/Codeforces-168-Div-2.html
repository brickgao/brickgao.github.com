D. Zero Tree

先使叶子节点为0，然后不断向上推出当前点操作，采用dfs。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 #define out(v) cerr << #v << ": " << (v) << endl
11 #define SZ(v) ((int)(v).size())
12 #define LL long long
13 const int maxint = -1u>>1;
14 template <class T> bool get_max(T& a, const T &b) {return b > a? a = b, 1: 0;}
15 template <class T> bool get_min(T& a, const T &b) {return b < a? a = b, 1: 0;}
16 
17 typedef struct record {
18     LL max, min;
19 }record;
20 
21 int n, u, v;
22 LL ans;
23 LL w[100010];
24 vector <int> p[100010];
25 
26 record dfs(int u, int fa) {
27     record tmp;
28     LL maxn = 0, minn = 0;
29     tmp.max = 0, tmp.min = 0;
30     for(int i = 0; i < SZ(p[u]); i ++) {
31         int v = p[u][i];
32         if(v != fa) {
33             tmp = dfs(v, u);
34             minn = min(minn, tmp.min);
35             maxn = max(maxn, tmp.max);
36         }
37     }
38     w[u] += maxn + minn;
39     tmp.min = minn;
40     tmp.max = maxn;
41     if(w[u] < 0) tmp.max -= w[u];
42     else tmp.min -= w[u];
43     //cout << u << " " << w[u] << " " << minn << " " << maxn << endl;
44     return tmp;
45 }
46 
47 int main() {
48     record tmp;
49     ans = 0;
50     cin >> n;
51     for(int i = 0; i < n; i ++) {
52         p[i].clear();
53     }
54     for(int i = 0; i < n - 1; i ++) {
55         cin >> u >> v;
56         p[u - 1].push_back(v - 1);
57         p[v - 1].push_back(u - 1);
58     }
59     for(int i = 0; i < n; i ++) {
60         cin >> w[i];
61     }
62     tmp = dfs(0, - 1);
63     ans = tmp.max - tmp.min;
64     cout << ans << endl;
65     return 0;
66 }
