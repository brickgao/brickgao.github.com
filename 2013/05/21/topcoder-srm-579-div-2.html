29         for(int j = i + 1; j < len; j ++) {
30             double tmp = sqrt(pow((double)(radius[i] + radius[j]), 2.0) - pow((double)abs(radius[i] - radius[j]), 2.0));

32         }
33     }

43 void dfs(int u, int vi, string st) {
44     if(vi == (1 << len) - 1) {
45         //cout << SZ(st) << endl;
46         double sum[11];
47         memset(sum, 0, sizeof(sum));
48         for(int i = 0; i < SZ(st); i ++) {
49             for(int j = 0; j < i; j ++) {
50                 sum[i] = max(sum[j] + map[(int)st[i]][(int)st[j]], sum[i]);
51             }
52         }
53         ans = min(ans, sum[len - 1]);
54         return;
55     }
56     for(int i = 0; i < len; i ++) {
57         if((vi & (1 << i)) == 0) {
58             string sttmp = st;
59             sttmp += (char)i;
60             dfs(i, (vi | (1 << i)), sttmp);
61         }
62     }
