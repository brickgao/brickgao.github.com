POJ-3026 Borg Maze

2012-09-04 #MST #Graph theory #POJ

题目传送门: POJ 3026

题目大概意思是#是墙，不能走，S是起点，A是目标点，要求从S出发走到所有点，路径不可交叉，然后求最小的路径和。

就是一个BFS+MST的题，自己BFS写的有问题，真是够糟糕的。。。

Code

  1 //By Brickgao
  2 #include <iostream>
  3 #include <cstdio>
  4 #include <cstring>
  5 #include <cmath>
  6 #include <cstdlib>
  7 #include <algorithm>
  8 #include <vector>
  9 #include <queue>
 10 
 11 using namespace std;
 12 
 13 typedef struct pos{
 14     int x, y;
 15     int step;
 16 } pos;
 17 pos p[120];
 18 char amap[120][120];
 19 bool vis[120], visit[120][120];
 20 long long n, x, y, k, ans, map[120][120], dist[120], num[120][120];
 21 long long add[4][2] = {
 22     {-1, 0}, 
 23     {0, -1}, 
 24     {0, 1}, 
 25     {1, 0}
 26 };
 27 
 28 void bfs(pos k)
 29 {
 30     int i, xx, yy, nx, ny, step;
 31     pos p;
 32     queue <pos> que;
 33     memset(visit, false, sizeof(visit));
 34     que.push(k);
 35     visit[k.x][k.y] = true;
 36     while(!que.empty())
 37     {
 38         p = que.front();
 39         que.pop();
 40         xx = p.x;
 41         yy = p.y;
 42         step = p.step + 1;
 43         for(i = 0; i < 4; i++)
 44         {
 45             nx = xx + add[i][0];
 46             ny = yy + add[i][1];
 47             if(nx < x && nx >= 0 && ny < y && ny >= 0 && !visit[nx][ny])
 48             {
 49                 if(amap[nx][ny] == 'S' || amap[nx][ny] == 'A')
 50                 {
 51                     map[num[k.x][k.y]][num[nx][ny]] = map[num[nx][ny]][num[k.x][k.y]] = step;
 52                     p.x = nx;
 53                     p.y = ny;
 54                     p.step = step;
 55                     que.push(p);
 56                     visit[nx][ny] = true;
 57                 }
 58                 if(amap[nx][ny] == ' ')
 59                 {
 60                     p.x = nx;
 61                     p.y = ny;
 62                     p.step = step;
 63                     que.push(p);
 64                     visit[nx][ny] = true;
 65                 }
 66             }
 67         }
 68     }
 69 }
 70 
 71 void prim(long long u)
 72 {
 73     long long i, j, min, tmp;
 74     for(i = 1; i <= k; i++)
 75     {
 76         dist[i] = map[u][i];
 77         vis[i] = false;
 78     }
 79     vis[u] = true;
 80     tmp = u;
 81     for(i = 1; i <= k; i++)
 82     {
 83         min = 0x7FFFFFFF;
 84         for(j = 1; j <= k; j++)
 85             if(!vis[j] && min > dist[j])
 86             {
 87                 tmp = j;
 88                 min = dist[j];
 89             }
 90         if(min != 0x7FFFFFFF)
 91             ans += min;
 92         vis[tmp] = true;
 93         for(j = 1; j <= k; j++)
 94             if(!vis[j] && dist[j] > map[tmp][j])
 95                 dist[j] = map[tmp][j];
 96     }
 97     return;
 98 }

100 int main()
101 {
102     long long i, j;
103     char tmp;
104     scanf("%lld", &n);
105     while(n--)
106     {
107         k = 1;
108         tmp = ' ';
109         scanf("%lld%lld", &x, &y);
110         while(tmp != '\n')
111             tmp = getchar();
112         for(j = 1; j <= y; j++)
113         {
114             for(i = 1; i <= x; i++)
115             {
116                 tmp = getchar();
117                 amap[i][j] = tmp;
118                 if(tmp == 'A' || tmp == 'S')
119                 {
120                     p[k].x = i;
121                     p[k].y = j;
122                     p[k].step = 0;
123                     num[i][j] = k;
124                     k++;
125                 }
126             }
127             getchar();
128         }
129         k--;
130         for(i = 1; i <= k; i++)
131             bfs(p[i]);
132         ans = 0;
133         prim(1);
134         printf("%lld\n", ans);
135     }
136     return 0;
137 }
