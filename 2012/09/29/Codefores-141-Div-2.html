B. Two Tables

给两个矩阵，定义一个运算∑ ai,j * bi+x,i+y,同时规定1 ≤ i ≤ na, 1 ≤ j ≤ ma, 1 ≤ i + x ≤ nb, 1 ≤ j + y ≤ mb。

求x, y为何值时，运算的值最大。

暴力模拟一遍就行，本来以为过不了了= =。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 int a[110][110], b[110][110];
12 int na, ma, nb, mb;
13 
14 int make(int x, int y)
15 {
16     int ans = 0;
17     for(int i = 1; i <= na; i++)
18         for(int j = 1; j <= ma; j++)
19         {
20             if(i + x > 0 && j + y > 0)
21             {
22                 ans += a[i][j] * b[i + x][j + y];
23             }
24         }
25     return ans;
26 }
27 
28 int main()
29 {
30     int recx, recy, max = -1;
31     char ch;
32     int mmax, nmax;
33     cin >> na >> ma;
34     for(int i = 1; i <= na; i++)
35     {
36         getchar();
37         for(int j = 1; j <= ma; j++)
38         {
39             ch = getchar();
40             a[i][j] = (int)ch - 48;
41         }
42     }
43     cin >> nb >> mb;
44     for(int i = 1; i <= nb; i++)
45     {
46         getchar();
47         for(int j = 1; j <= mb; j++)
48         {
49             ch = getchar();
50             b[i][j] = (int)ch - 48;
51         }
52     }
53     if(ma > mb) mmax = ma; else mmax = mb;
54     if(na > nb) nmax = na; else nmax = nb;
55     for(int i = 1 - nmax; i <= nmax - 1; i++)
56         for(int j = 1 - mmax; j <= mmax - 1; j++)
57         {
58             int tmp = make(i, j);
59             if(tmp > max)
60             {
61                 recx = i;
62                 recy = j;
63                 max = tmp;
64             }
65         }
66     printf("%d %d\n", recx, recy);
