A. Dice Tower

喜闻乐见的骰子题，将骰子累起来，两个接触的面值不同，一个骰子两个相对的面和为7，每一个骰子判断就行了。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 
12 int main()
13 {
14     int n, x, a, b, tmp;
15     bool flag;
16     while(cin >> n >> x)
17     {
18         flag = true;
19         cin >> a >> b;
20         tmp = 7 - x;
21         for(int i = 2; i <= n; i++)
22         {
23             cin >> a >> b;
24             if(a == tmp || b == tmp || 7 - a == tmp || 7 - b == tmp || a == x || b == x || 7 - a == x || 7 - b == x)
25             {
26                 flag = false;
27             }
28         }
29         if(flag) printf("YES\n");
30         else printf("NO\n");
31     }
32 }

B. Well-known Numbers

一个类fibonacci数列,求出序列，然后倒序贪心求和位s就行。

注意求出序列的每个数不同O_o

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 #define LL long long
10 using namespace std;
11 
12 LL f[1000], k, s;
13 
14 int main()
15 {
16     vector <LL> rec;
17     LL i;
18     cin >> s >> k;
19     f[0] = 0;
20     f[1] = 1;
21     for(i = 2; f[i - 1] < s; i++)
22         for(LL j = i - 1; j >= 0 && j >= i - k; --j)
23             f[i] += f[j];
24     for(; i > 0 && s > 0; i--)
25         if(f[i] <= s)
26         {
27             s -= f[i];
28             rec.push_back(f[i]);
29         }
30     LL leng = rec.size();
31     cout << leng << endl;
32     for(i = 0; i < leng - 1; i++)
33         cout << rec[i] << " ";
34     cout << rec[leng - 1] << endl;
35     return 0;
36 }
