POJ-1753 Flip Game

2012-09-13 #Enumeration #POJ

题目传送门: POJ 1753

给一个4 * 4的网格，每个格子中放白棋子或者黑棋子。

定义一个翻转操作，白子变黑子，黑子变白子。

每次可以对棋子进行一个操作，若操作的棋子是第i行第j列，则(i,j) (i - 1, j) (i + 1, j) (i, j - 1) (i, j +1)均进行翻转，越界的棋子就不翻转了。

最后问操作至全白或全黑的最小操作数。

具体就是一个不难的枚举题。

具体枚举方法是先对第一行进行操作（共16种）。

以后每一行操作，使上一行全为白或全为黑。

最后如果最后一行不满足全为白或黑，就无法达到操作要求，否则可以达到要求。

Code

  1 //By Brickgao
  2 #include <iostream>
  3 #include <cstdio>
  4 #include <cstring>
  5 #include <cmath>
  6 #include <cstdlib>
  7 #include <algorithm>
  8 #include <vector>
  9 using namespace std;
 10 
 11 char rec[6][6], map[6][6];
 12 int num, wdo[20], bdo[20];
 13 
 14 int oneflip(int a, int b)
 15 {
 16     if(rec[a][b] == 'b') rec[a][b] = 'w';
 17     else rec[a][b] = 'b';
 18     return 0;
 19 }
 20 
 21 int flip(int a, int b)
 22 {
 23     oneflip(a, b);
 24     oneflip(a - 1, b);
 25     oneflip(a + 1, b);
 26     oneflip(a, b + 1);
 27     oneflip(a, b - 1);
 28     return 0;
 29 }
 30 
 31 int makeblack()
 32 {
 33     int ans = 0, flag = false;
 34     for(int i = 2; i <= 4; i++)
 35         for(int j = 1; j <= 4; j++)
 36             if(rec[i - 1][j] == 'w')
 37             {
 38                 flip(i, j);
 39                 ans ++;
 40             }
 41     for(int i = 1; i <= 4; i++)
 42         if(rec[4][i] == 'w')
 43             flag = true;
 44     if(flag)
 45         ans = 9999;
 46     return ans;    
 47 }
 48 
 49 int makewhite()
 50 {
 51     int ans = 0, flag = false;
 52     for(int i = 2; i <= 4; i++)
 53         for(int j = 1; j <= 4; j++)
 54             if(rec[i - 1][j] == 'b')
 55             {
 56                 flip(i, j);
 57                 ans ++;
 58             }
 59     for(int i = 1; i <= 4; i++)
 60         if(rec[4][i] == 'b')
 61             flag = true;
 62     if(flag)
 63         ans = 9999;
 64     return ans;    
 65 }
 66 
 67 void reset()
 68 {
 69     for(int i = 0; i <= 5; i++)
 70         for(int j = 0; j <= 5; j++)
 71             rec[i][j] = map[i][j];
 72 }
 73 
 74 int main()
 75 {
 76     int ans, num = 0;
 77     memset(map, 'b', sizeof(map));
 78     for(int i = 1; i <= 4; i++)
 79     {
 80         for(int j = 1; j <= 4; j++)
 81             scanf("%c", &map[i][j]);
 82         getchar();
 83     }
 84     reset();
 85     bdo[num] = makeblack();
 86     reset();
 87     wdo[num] = makewhite();
 88     num ++;
 89     for(int i = 1; i <= 4; i ++)
 90     {
 91         reset();
 92         flip(1, i);
 93         bdo[num] = makeblack() + 1;
 94         reset();
 95         flip(1, i);
 96         wdo[num] = makewhite() + 1;
 97         num ++;
 98     }
 99     for(int i = 1; i <= 3; i ++)
100         for(int j = i + 1; j <= 4; j ++)
101         {
102             reset();
103             flip(1, i);
104             flip(1, j);
105             bdo[num] = makeblack() + 2;
106             reset();
107             flip(1, i);
108             flip(1, j);
109             wdo[num] = makewhite() + 2;
110             num ++;
111         }
112     for(int i = 1; i <= 4; i ++)
113     {
114         reset();
115         for(int j = 1; j <= 4; j++)
116             if(j != i)            
117                 flip(1 , j);
118         bdo[num] = makeblack() + 3;
119         reset();
120         for(int j = 1; j <= 4; j++)
121             if(j != i)
122                 flip(1 , j);
123         wdo[num] = makewhite() + 3;
124         num ++;
125     }
126     reset();
127     for(int i = 1; i <= 4; i ++)
128         flip(1, i);
129     wdo[num] = makewhite() + 4;
130     reset();
131     for(int i = 1; i <= 4; i ++)
132         flip(1, i);
133     bdo[num] = makeblack() + 4;
134     ans = 9999;
135     for(int i = 0; i <= 15; i++)
136     {
137         if(ans > bdo[i]) ans = bdo[i];
138         if(ans > wdo[i]) ans = wdo[i];
139     }
140     if(ans >= 9999) printf("Impossible\n");
141     else
142         printf("%d\n", ans);
143     return 0;
144 }
