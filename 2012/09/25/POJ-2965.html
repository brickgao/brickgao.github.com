POJ-2965 The Pilots Brothers‘ refrigerator

2012-09-25 #Enumeration #POJ

题目传送门: POJ 2965

给一个矩阵，矩阵中只有“+”和“-”，“+”代表关闭的开关，“-”代表打开的开关，现在需要把所有的开关打开。

每次改变一个开关的状态，就同时会改变同一行或者同一列的开关的状态。

使用枚举就可以了，运用二进制来表示开关的状态=w=

Code

  1 //By Brickgao
  2 #include <iostream>
  3 #include <cstdio>
  4 #include <cstring>
  5 #include <cmath>
  6 #include <cstdlib>
  7 #include <algorithm>
  8 #include <vector>
  9 using namespace std;
 10 
 11 char map[6][6], op[6][6];
 12 int x[16], y[16], total;
 13 
 14 int reset()
 15 {
 16     for(int i = 0; i <= 3; i++)
 17         for(int j = 0; j <= 3; j++)
 18             op[i][j] = map[i][j];
 19     return 0;
 20 }
 21 
 22 int make(int a, int b)
 23 {
 24     for(int i = 0; i <= 3; i++)
 25     {
 26         if(op[a][i] == '-')
 27             op[a][i] = '+';
 28         else
 29             op[a][i] = '-';
 30     }
 31     for(int i = 0; i <= 3; i++)
 32     {
 33         if(op[i][b] == '-')
 34             op[i][b] = '+';
 35         else
 36             op[i][b] = '-';
 37     }
 38     if(op[a][b] == '-')
 39         op[a][b] = '+';
 40     else
 41         op[a][b] = '-';
 42     return 0;
 43 }
 44 
 45 int main()
 46 {
 47     int rec, temp, row, col, t;
 48     bool flag1 = false, flag2;
 49     for(int i = 0; i <= 3; i++)
 50     {
 51         for(int j = 0; j <= 3; j++)
 52             map[i][j] = getchar();
 53         getchar();
 54     }
 55     for(int i = 0; i <= 1 << 16; i++)
 56     {
 57         flag2 = true;
 58         reset();
 59         temp = i;
 60         for(int j = 0; j < 16; j++)
 61         {
 62             if((temp & 1) == 1)
 63             {
 64                 col = j / 4;
 65                 row = j % 4;
 66                 make(row, col);
 67             }
 68             temp = temp >> 1;
 69         }
 70         for(int j = 0; j <= 3; j++)
 71             for(int k = 0; k <= 3; k++)
 72             {
 73                 if(op[j][k] == '+')
 74                 {
 75                     flag2 = false;
 76                     break;
 77                 }
 78             }
 79         if(flag2)
 80         {
 81             flag1 = true;
 82             rec = i;
 83             break;
 84         }
 85     }
 86     if(flag1)
 87     {
 88         t = 0;
 89         temp = rec;
 90         for(int i = 0; i < 16; i++)
 91         {
 92             if((temp & 1) == 1)
 93             {
 94                 col = i / 4;
 95                 row = i % 4;
 96                 x[t] = row;
 97                 y[t ++] = col;
 98             }
 99             temp = temp >> 1;
100         }
101         printf("%d\n", t);
102         for(int i = 0; i < t; i++)
103             printf("%d %d\n", x[i] + 1, y[i] + 1);
104     }
105     return 0;
106 }
