A. Shooshuns and Sequence

简单题，给N个数，每次进行两个操作，把第K个数复制至最后一个，然后删除第一个数，最后要求一行数全部相同。

具体来说只要第K个数后的数完全相同，最后就可以达到题目所要求的。

 1 //by Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstdlib>
 5 #include <cstring>
 6 #include <string>
 7 
 8 using namespace std;
 9 
10 __int64 a[100100];
11 
12 int main()
13 {
14     __int64 n, k, ans, i;
15     bool flag;
16     while(~scanf("%I64d%I64d", &n, &k))
17     {
18         for(i = 1; i <= n; i++)
19             scanf("%I64d", &a[i]);
20         flag = true;
21         for(i = k + 1; i <= n; i++)
22             if(a[k] != a[i])
23             {
24                 flag = false;
25                 break;
26             }
27         if(!flag)
28         {
29             printf("-1\n");
30         }
31         else
32         {
33             for(i = k - 1; i >= 1; i --)
34                 if(a[i] != a[k])
35                     break;
36             ans = i;
37             printf("%I64d\n", ans);
38         }
39     }
40     return 0;
41 }

B. Cosmic Tables

大意就是对一个矩阵进行操作。

用两个数组记录行号列号直接进行操作就行。

 1 //by Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstdlib>
 5 #include <cstring>
 6 #include <string>
 7 
 8 using namespace std;
 9 
10 int n, m, k;
11 int row[1010], col[1010];
12 int rec[1010][1010];
13 
14 int swaprow(int a, int b)
15 {
16     int tmp;
17     tmp = row[a];
18     row[a] = row[b];
19     row[b] = tmp;
20     return 0;
21 }
22 
23 int swapcol(int a, int b)
24 {
25     int tmp;
26     tmp = col[a];
27     col[a] = col[b];
28     col[b] = tmp;
29     return 0;
30 }
31 
32 int main()
33 {
34     char ch;
35     int tmp1, tmp2, i, j;
36     while(~scanf("%d%d%d", &n, &m, &k))
37     {
38         for(i = 1; i <= n; i++)
39             row[i] = i;
40         for(i = 1; i <= m; i++)
41             col[i] = i;
42         for(i = 1; i <= n; i++)
43             for(j = 1; j <= m; j++)
44                 scanf("%d", &rec[i][j]);
45         for(i = 1; i <= k; i++)
46         {
47             getchar();
48             scanf("%c %d %d", &ch, &tmp1, &tmp2);
49             if(ch == 'r') swaprow(tmp1, tmp2);
50             if(ch == 'c') swapcol(tmp1, tmp2);
51             if(ch == 'g') printf("%d\n", rec[row[tmp1]][col[tmp2]]);
52         }
53     }
54     return 0;
55 }

D. Olympiad

给一个最小分数，再给两组分，每个人取一个第一组的分，再取一个第二组的分，求这个人的最小名次和最大名次。

比较简单的贪心，但是我贪心策略一开始写错了=A=
