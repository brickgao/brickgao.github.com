550pt BinaryCode

给一种二进制串加密模式，串头部等于串头部数字加后一个数字，串尾部数字等于尾部数字加上前一个数字，其他位置上的数字的前一个数字加上后一个数字。

然后假设第一位是1或者0求是否能解密出串，解密不出输出“NONE”。

 1 #include <cstdio>
 2 #include <iostream>
 3 #include <cstring>
 4 #include <algorithm>
 5 #include <cmath>
 6 #include <vector>
 7 using namespace std ;
 8 #define For(i , n) for(int i = 0 ; i < (n) ; ++i)
 9 #define SZ(x)  (int)((x).size())
10 typedef long long lint ;
11 const int maxint = -1u>>2 ;
12 const double eps = 1e-6 ;
13  
14 
15 class BinaryCode
16 {
17 public:
18 vector <string> decode(string message)
19 {
20  vector <string> ans;
21  bool flag;
22  string st;
23  int tmp;
24  if(message.size() != 1)
25  {
26      flag = true;
27      st.clear();
28      st += '0';
29      for(int i = 0; i < message.size() - 1; i ++)
30      {
31          if(i != 0)
32              tmp = message[i] - st[i] - (st[i - 1] - '0');
33          else
34              tmp = message[i] - st[i];
35          if(tmp != 0 && tmp != 1)    flag = false;
36          st += '0' + tmp;
37      }
38      if(message[message.size() - 1] == st[st.size() - 1] + st[st.size() -2] - '0' && flag)
39          ans.push_back(st);
40      else
41          ans.push_back("NONE");
42      st.clear();
43      flag = true;
44      st += '1';
45      for(int i = 0; i < message.size() - 1; i ++)
46      {
47          if(i != 0)
48              tmp = message[i] - st[i] - (st[i - 1] - '0');
49          else
50              tmp = message[i] - st[i];
51          if(tmp != 0 && tmp != 1) flag = false;
52          st += '0' + tmp;
53      }
54      if(message[message.size() - 1] == st[st.size() - 1] + st[st.size() -2] - '0' && flag)
55          ans.push_back(st);
56      else
57          ans.push_back("NONE");
58  }
59  else
60  {
61      if(message[0] == '0')
62          ans.push_back("0");
63      else
64          ans.push_back("NONE");
65      if(message[0] == '1')
66          ans.push_back("1");
67      else
68          ans.push_back("NONE");
69 
70  }
71  return vector <string>(ans) ;
72 }
73 };

1100pt PowerOutage

给一个图，求从root遍历所有点的费用。

树形DP...第一个树形dp【dp还没搞清楚 = =。

全部路径的费用 * 2减去最大的从某一点到root的距离。

 1 #include <cstdio>
 2 #include <iostream>
 3 #include <cstring>
 4 #include <algorithm>
 5 #include <cmath>
 6 #include <vector>
 7 #include <stack>
 8 using namespace std ;
 9 #define For(i , n) for(int i = 0 ; i < (n) ; ++i)
10 #define SZ(x)  (int)((x).size())
11 typedef long long lint ;
12 const int maxint = -1u>>2 ;
13 const double eps = 1e-6 ;
14 
15 class PowerOutage
16 {
17 public:
18 int n;
19 int map[100][100];
20 
21 int dfs(int pre, int cur)
22 {
23  int ans = 0;
24  for(int i = 0; i <= n; i ++)
25      if(map[cur][i] != -1 && i != pre)
26      {
27          int tmp = dfs(cur, i);
28          if(tmp > ans) ans = tmp;
29      }
30  if(cur != 0) ans += map[pre][cur];
31  return ans;
32 }
33 
34 int estimateTimeOut(vector <int> fromJunction, vector <int> toJunction, vector <int> ductLength)
35 {
36  int ans = 0;
37  n = 0;
38  memset(map, -1, sizeof(map));
39  for(int i = 0; i < fromJunction.size(); i ++)
40  {
41      map[fromJunction[i]][toJunction[i]] = map[toJunction[i]][fromJunction[i]] = ductLength[i];
42      ans += ductLength[i];
43      if(n < max(fromJunction[i], toJunction[i]))
44              n = max(fromJunction[i], toJunction[i]);
45  }
46  ans *= 2;
47  ans -= dfs(-1, 0);
48  return (int)ans;
49 }
50 };
