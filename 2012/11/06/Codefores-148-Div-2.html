A. Two Bags of Potatoes

两袋子土豆，一袋有x个，另一袋有y个，（x + y）不大于n，且是k的倍数。

现在知道y以及n，k，求所有可能的x。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 int main()
12 {
13     long long y, k, n;
14     bool flag = true;
15     cin >> y >> k >> n;
16     for(long long i = 0; i <= n; i += k)
17     {
18         if(flag)
19         {
20             if(i > y)
21             {
22                 cout << i - y;
23                 flag = false;
24             }
25         }
26         else
27         {
28             if(i > y)   cout << " " << i - y;
29         }
30     }
31     if(flag) cout << "-1";
32     cout << endl;
33     return 0;
34 }

B. Easy Tape Programming

介绍了一种新的语言，分别有一个指针和一个方向。

最初指针指向字符串的最左边，方向为右。

有指针指向方向和数字代表两种操作。

如果指向方向字符(">", "<")改变相应的方向，然后向相应的方向移动一位，如果下一位还是方向字符，则删除前一字符。

如果指向数字，则输出该位数字，然后再向对应方向移动一位，前一位数字减一，如果前一位为0，则删除前一字符。

进行操作至超出边界。

现在给出字符串以及l，r。对sl ~ sr的字串进行操作，求各个数字输出的个数。

模拟即可。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 typedef string ::iterator it_t;
12 
13 int main()
14 {
15     __int64 n, q, ans[10], now, l, r;
16     it_t it;
17     string s, stmp;
18     cin >> n >> q;
19     cin >> s;
20     for(__int64 i = 1; i <= q; i ++)
21     {
22         bool turn = true;
23         stmp = s;
24         memset(ans, 0, sizeof(ans));
25         cin >> l >> r;
26         l --;
27         r --;
28         now = l;
29         while(now <= r && now >= l)
30         {
31             if(stmp[now] == '>')
32             {
33                 turn = true;
34                 now ++;
35                 if(now <= r && now >= l)
36                     if(stmp[now] == '>' || stmp[now] == '<')
37                     {
38                         it = stmp.begin() + now - 1;
39                         stmp.erase(it);
40                         r --;
41                         now --;
42                     }
43                 continue;
44             }
45             else if(stmp[now] == '<')
46                 {
47                     turn = false;
48                     now --;
49                     if(now <= r && now >= l)
50                         if(stmp[now] == '>' || stmp[now] == '<')
51                         {
52                             it = stmp.begin() + now + 1;
53                             stmp.erase(it);
54                             r --;
55                         }
56                     continue;
57                 }
58                 else
59                 {
60                     ans[stmp[now] - 48] ++;
61                     if(turn)
62                     {
63                         now ++;
64                         if(stmp[now - 1] == '0')
65                         {
66                             it = stmp.begin() + now - 1;
67                             stmp.erase(it);
68                             r --;
69                             now --;
70                         }
71                         else
72                             stmp[now - 1] = stmp[now - 1] - 1;
73                     }
74                     else
75                     {
76                         now --;
77                         if(stmp[now + 1] == '0')
78                         {
79                             it = stmp.begin() + now + 1;
80                             stmp.erase(it);
81                             r --;
82                         }
83                         else
84                             stmp[now + 1] = stmp[now + 1] - 1;
85                     }
86                 }
87         }
88         for(__int64 j = 0; j <= 9; j ++)
89             if(j)
90                 printf(" %I64d", ans[j]);
91             else
92                 printf("%I64d", ans[j]);
93         cout << endl;
94     }
95     return 0;
96 }

C. Not Wool Sequences

给两个数字n，m。

从0 - 2m - 1中选择n个数字组成一个数列。要求数列中存在一个子数列从左到右进行异或操作的答案为0。

求不符合答案的数列个数。

找规律Orz。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 #define modulo 1000000009
10 using namespace std;
11 
12 int main()
13 {
14     long long n, m, ans = 1, tmp = 1;
15     cin >> n >> m;
16     while(m --)
17     {
18         tmp <<= 1;
19         tmp %= modulo;
20     }
21     for(long long i = 1; i <= n; i ++)
22     {
23         ans *= abs(tmp - i);
24         ans %= modulo;
25     }
26     ans = (ans + modulo) % modulo;
27     cout << ans << endl;
28     return 0;
29 }
