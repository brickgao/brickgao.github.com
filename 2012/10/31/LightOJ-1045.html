LightOJ-1045 Digits of Factorial

2012-10-31 #Math #LightOJ

题目传送门: LightOJ 1045

就是求n！在base进制下有多少位。

数论题，查到了斯特林[striling]公式这个东西：lnN!=NlnN－N+0.5ln(2N*pi)

所以base进制下的位数为[(lnN!/ln(base))]

数学真是神奇=-=

Code

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 int main()
12 {
13     long long t, base, caseno = 1, ans;
14     double tmp, n;
15     scanf("%lld", &t);
16     while(t--)
17     {
18         scanf("%lf%lld", &n, &base);
19         if(n != 0)
20         {
21             tmp = n * log(n) - n + 0.5 * log(4 * n * acos(0.0));
22             tmp = tmp / log(base);
23             ans = tmp;
24             if(tmp - (double)ans > 0) ans++;
25         }
26         else
27             ans = 1;
28         printf("Case %lld: %lld\n", caseno ++, ans);
29     }
30     return 0;
31 }
