 12 const int Maxn = 18000;
 13 const int Maxm = 220000;
 14 
 15 typedef struct Edge{
 16     int to, nxt;
 17 }Edge;
 18 
 19 Edge rec[Maxm];
 20 int head[Maxn], dist[Maxn], que[Maxm];
 21 bool visit[Maxn], forbid[Maxn];
 22 int cnt = 0, caseno = 1, start, end;
 23 string a, b;
 24 
 25 void add(int u, int v)
 26 {
 27     rec[cnt].nxt = head[u];
 28     rec[cnt].to = v;
 29     head[u] = cnt ++;
 30     rec[cnt].nxt = head[v];
 31     rec[cnt].to = u;
 32     head[v] = cnt ++;
 33 }
 34 
 35 int get_num(string s)
 36 {
 37     int ans = s[0] - 'a' + (s[1] - 'a') * 26 + (s[2] - 'a') * 26 * 26;
 38     return ans;
 39 }
 40 
 41 void reset_graph()
 42 {
 43     for(int i = 0; i < 26; i ++)
 44         for(int j = 0; j < 26; j ++)
 45             for(int k = 0; k < 26; k ++)
 46             {
 47                 int num1, num2;
 48                 string tmp1 = "";
 49                 tmp1 += 'a' + i; tmp1 += 'a' + j; tmp1 += 'a' + k;
 50                 num1 = get_num(tmp1);
 51                 for(int p = 0; p < 3; p ++)
 52                 {
 53                     string tmp2 = tmp1;
 54                     tmp2[p] --;
 55                     if(tmp2[p] < 'a')   tmp2[p] = 'z';
 56                     num2 = get_num(tmp2);
 57                     add(num1, num2);
 58                     tmp2 = tmp1;
 59                     tmp2[p] ++;
 60                     if(tmp2[p] > 'z')   tmp2[p] = 'a';
 61                     num2 = get_num(tmp1);
 62                     add(num1, num2);
 63                 }
 64             }
 65 }

 67 void spfa()
 68 {
 69     int h = 0 , f = 0 ;
 70     que[++ f] = start ; dist[start] = 0 ;
 71     visit[start] = true;
 72     while (h <= f)
 73     {
 74         int u = que[++ h];
 75         for(int i = head[u]; i != -1; i = rec[i].nxt)
 76         {
 77             int v = rec[i].to;
 78             if (forbid[v]) continue;
 79             if (dist[u] + 1 < dist[v])
 80             {
 81                 dist[v] = dist[u] + 1;
 82                 if (!visit[v])
 83                 {
 84                     que[++ f] = v;
 85                     visit[v] = true;
 86                 }
 87             }
 88             visit[u] = false;
 89         }
 90     }
 91 }
 92 
 93 void solve()
 94 {
 95     if(forbid[start] || forbid[end])
 96     {
 97         printf("Case %d: -1\n", caseno ++);
 98         return;
 99     }
100     spfa();
101     if(dist[end] != 32766)
102         printf("Case %d: %d\n", caseno ++, dist[end]);
103     else
104         printf("Case %d: -1\n", caseno ++);
105 }

107 void init()
108 {
109     char s1[26], s2[26], s3[26];
110     int n;
111     memset(dist, Maxn, sizeof(dist));
112     memset(visit, false, sizeof(visit));
113     memset(forbid, false, sizeof(forbid));
114     cin >> a >> b;
115     start = get_num(a);
116     end = get_num(b);
117     scanf("%d", &n);
118     for(int i = 0; i < n; i ++)
119     {
120         scanf("%s %s %s", s1, s2, s3);
121         for(int ii = 0; ii < strlen(s1); ii ++)
122             for(int jj = 0; jj < strlen(s2); jj ++)
123                 for(int kk = 0; kk < strlen(s3); kk++)
124                 {
125                     string tmp = "";
126                     tmp += s1[ii]; tmp += s2[jj]; tmp += s3[kk];
127                     forbid[get_num(tmp)] = true;
128                 }
129     }
130 }
131 
132 int main()
133 {
134     int t;
135     memset(head, -1, sizeof(head));
136     reset_graph();
137     scanf("%d", &t);
138     while(t --)
139     {
140         init();
141         solve();
142     }
143     return 0;
144 }

从Forum学习了一下如何一颗树上最远的两个点，每次求以该点为根的最大点和次大点处理。

  1 //By Brickgao
  2 #include <iostream>
  3 #include <cstdio>
  4 #include <cstring>
  5 #include <cmath>
  6 #include <cstdlib>
  7 #include <algorithm>
  8 #include <vector>
  9 using namespace std;
 10 
 11 #define pb push_back
 12 #define cl clear
 13 
 14 typedef struct record{
 15     int to, w;
 16 }record;
 17 
 18 int t, caseno = 1, n, ans, dist[30010][2];
 19 bool visit[30010];
 20 vector <record> p[30010];
 21 
 22 void dfs(int u)
 23 {
 24     bool done = false;
 25     int max1 = 0, max2 = 0;
 26     for(int i = 0; i < p[u].size(); i ++)
 27     {
 28         int v = p[u][i].to;
 29         if(!visit[v])
 30         {
 31             done = true;
 32             visit[v] = true;
 33             dfs(v);
 34             int total = max(dist[v][0], dist[v][1]);
 35             //cout << u << "->" << v << " " << dist[v] + p[u][i].w << endl;
 36             if(total + p[u][i].w > max2 && total + p[u][i].w <= max1)
 37                 max2 = total + p[u][i].w;
 38             if(total + p[u][i].w > max1)
 39             {
 40                 max2 = max1;
 41                 max1 = total + p[u][i].w;
 42                 //cout << "change: " << max1 << endl;
 43             }
 44         }
 45     }
 46     //cout << max1 << "***" << max2 << endl;
 47     if(done)
 48     {
 49         if(max2 == 0)
 50             dist[u][0] = max1;
 51         else
 52         {
 53             dist[u][0] = max1;
 54             dist[u][1] = max2;
 55         }
 56     }
 57 }
 58 
 59 void init()
 60 {
 61     int u, v, w;
 62     record tmp;
 63     ans = 0;
 64     scanf("%d", &n);
 65     for(int i = 0; i < n; i ++)
 66         p[i].cl();
 67     for(int i = 0; i < n - 1; i ++)
 68     {
 69         scanf("%d%d%d", &u, &v, &w);
 70         tmp.w = w;
 71         tmp.to = v; p[u].pb(tmp);
 72         tmp.to = u; p[v].pb(tmp);
 73     }
 74 }
 75 
 76 void solve()
 77 {
 78     for(int i = 0; i <= n; i ++)
 79     {
 80         visit[i] = false;
 81         dist[i][0] = dist[i][1] = 0;
 82     }
 83     visit[0] = true;
 84     dfs(0);
 85     for(int i = 0; i < n; i ++)
 86     {
 87         //cout << dist[i] << endl;
 88         ans = max(ans, dist[i][0] + dist[i][1]);
 89     }
 90     printf("Case %d: %d\n", caseno ++, ans);
 91 }
 92 
 93 int main()
 94 {
 95     scanf("%d", &t);
 96     while(t --)
 97     {
 98         init();
 99         solve();
100     }
101     return 0;
102 }
