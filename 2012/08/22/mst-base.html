 11 typedef struct lin{
 12     int a;
 13     int b;
 14     int len;
 15 } lin;
 16 
 17 int n, map[110][110];
 18 lin le[225000];
 19 int parent[225000], q;
 20 int rank[225000];
 21 
 22 void makeset(int n)
 23 {
 24     for(int i = 0; i < n; i++)
 25     {
 26         parent[i] = i;
 27         rank[i] = 0;
 28     }
 29 }
 30 
 31 int getparent(int x)
 32 {
 33     if(parent[x] == x)
 34         return x;
 35     else
 36     {
 37         parent[x] = getparent(parent[x]);
 38         return parent[x];
 39     }
 40 }
 41 
 42 bool cmp(lin p, lin q)
 43 {
 44     return p.len < q.len;
 45 }
 46 
 47 void unionset(int x, int y)
 48 {
 49     if(rank[x] > rank[y])
 50         parent[y] = x;
 51     else
 52     {
 53         parent[x] = y;
 54         if(rank[x] == rank[y])
 55             ++rank[y];
 56     }
 57 }
 58 
 59 int main()
 60 {
 61     int a, b, ans = 0, i, j, count, k;
 62     while(~scanf("%d", &n))
 63     {
 64         for(i = 1; i <= n; i++)
 65             for(j = 1; j <= n; j++)
 66                 scanf("%d", &map[i][j]);
 67         scanf("%d", &q);
 68         for(i = 0; i < q; i++)
 69         {
 70             scanf("%d%d", &a, &b);
 71             map[a][b] = map[b][a] = 0;
 72         }
 73         k = 0;
 74         for(i = 1; i <= n; i++)
 75             for(j = 1; j <= n; j++)
 76                 if(j < i)
 77                 {
 78                     le[k].a = i;
 79                     le[k].b = j;
 80                     le[k++].len = map[i][j];
 81                 }
 82         sort(le, le + k, cmp);
 83         makeset(n);
 84         count = 0;
 85         for(i = 0; i < k; i++)
 86         {
 87             int x = getparent(le[i].a);
 88             int y = getparent(le[i].b);
 89             if(x != y)
 90             {
 91                 unionset(x, y);
 92                 count++;
 93                 ans += le[i].len;
 94                 if(count == n - 1)
 95                     break;
 96             }
 97         }
 98         printf("%d\n", ans);
 99         q = 0;
100         count = 0;
101         ans = 0;
102     }
103     return 0;

poj 1094 Sorting It All Out

基本拓扑排序，具体拓扑排序就是每次找出图中入度为零的点，然后删去与其相连的边，再寻找新的入度为零的点，不断处理，找出点的顺序就是拓扑排序的顺序。

 1 //By Brickgao
 2 #include <iostream>
 3 #include <cstdio>
 4 #include <cstring>
 5 #include <cmath>
 6 #include <cstdlib>
 7 #include <algorithm>
 8 #include <vector>
 9 using namespace std;
10 
11 int map[50][50], t[50], d[50], ans[50];
12 int n, m, flag1;
13 char s[10];
14 
15 int topu()
16 {
17  int i, j, flag2 = 1, p, tmp, k = 0;
18  for(i = 0; i < n; i++)
19      t[i] = d[i];
20  for(i = 0; i < n; i++)
21  {
22      p = 0;
23      for(j = 0; j < n; j++)
24          if(t[j] == 0)
25          {
26              p++;
27              tmp = j;
28          }
29      if(p == 0 && p != n) return 0;
30      if(p > 1) flag2 = -1;
31      ans[k++] = tmp;
32      t[tmp] = -1;
33      for(j = 0; j < n; j++)
34          if(map[tmp][j] == 1)
35              t[j] --;
36  }
37  return flag2;
38 }
39 
40 int main()
41 {
42  int i, p, j;
43  while(~scanf("%d%d", &n, &m))
44  {
45      if(m == 0 && n == 0) break;
46      flag1 = 0;
47      memset(map, 0, sizeof(map));
48      memset(d, 0, sizeof(d));
49      for(i = 0; i < m; i++)
50      {
51          scanf("%s", s);
52          map[s[0] - 'A'][s[2] - 'A'] = 1;
53          d[s[2] - 'A']++;
54          if(flag1 == 0)
55              p = topu();
56          if(p == 1 && flag1 == 0)
57          {
58              printf("Sorted sequence determined after %d relations: ", i + 1);
59              for(j = 0; j < n; j++)
60                  printf("%c", ans[j] + 'A');
61              printf(".\n");
62              flag1 = 1;
63          }
64          if(p == 0 && flag1 == 0)
65          {
66              printf("Inconsistency found after %d relations.\n", i + 1);
67              flag1 = 1;
68          }
69      }
70      if(flag1 == 0)
71          printf("Sorted sequence cannot be determined.\n");
72  }
73     return 0;
74 }
